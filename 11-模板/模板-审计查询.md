## 一、KP-题型-题库三角验证（Agent 9 · #17）

> **逻辑**：检查每个 KP 声明的 `problem_types` 是否在题库中有实际题目支撑。
> 由于 Dataview 无法直接跨 frontmatter 数组做精确匹配，以下采用「近似版」+「精确版」两种策略。

### 1.1 近似版（Dataview，快速筛查）

检查"声明了题型的 KP 数量" vs "实际关联了题库文件的 KP 数量"。

```dataview
TABLE
  length(file.outlinks) AS 出链总数,
  length(filter(file.outlinks, (l) => contains(string(l), "04-题库"))) AS 题库链接数,
  length(problem_types) AS 声明题型数,
  choice(length(filter(file.outlinks, (l) => contains(string(l), "04-题库"))) > 0, "✅", "❌") AS 有题库关联
FROM "03-知识点"
WHERE type = "知识点"
  AND length(problem_types) > 0
SORT 题库链接数 ASC, 声明题型数 DESC
```

### 1.2 精确版（DataviewJS，逐题型核对）

```dataviewjs
// KP-题型-题库三角验证（精确版）
// 逻辑：对每个 KP，检查其 problem_types 中每个题型在题库中的出现次数

const kpPages = dv.pages('"03-知识点"').where(p => p.type == "知识点" && p.problem_types);
const problemPages = dv.pages('"04-题库"').where(p => p.type == "题目");

// 构建 题型 -> 题目数 映射（基于 question_type 和 knowledge_points）
const typeCount = {};
const kpLinkCount = {}; // 知识点名 -> 引用该知识点的题目数

for (const q of problemPages) {
  // 按 question_type 统计
  const qt = q.question_type;
  if (qt) {
    typeCount[qt] = (typeCount[qt] || 0) + 1;
  }
  // 按 knowledge_points 统计（建立 KP 名 -> 题目数映射）
  const kps = q.knowledge_points || [];
  for (const k of kps) {
    kpLinkCount[k] = (kpLinkCount[k] || 0) + 1;
  }
}

const results = [];
for (const kp of kpPages) {
  const pts = kp.problem_types || [];
  const kpName = kp.file.name;
  const linkedCount = kpLinkCount[kpName] || 0;

  // 问题判定：声明了题型，但 knowledge_points 中无题目引用该 KP
  if (pts.length > 0 && linkedCount == 0) {
    // 进一步检查每个题型的全局题目数
    const typeDetails = pts.map(pt => {
      const cleanType = pt.replace("题型-", "");
      const count = typeCount[cleanType] || 0;
      return `${pt}(${count})`;
    }).join(", ");

    results.push({
      kp: `[[${kp.file.path}|${kpName}]]`,
      types: typeDetails,
      linked: linkedCount,
      issue: "声明了题型但无题目关联该KP"
    });
  }
}

if (results.length === 0) {
  dv.paragraph("✅ 所有声明题型的 KP 均有题库题目关联（通过 knowledge_points）");
} else {
  dv.table(
    ["知识点", "声明题型(题库数)", "关联题目数", "问题"],
    results.map(r => [r.kp, r.types, r.linked, r.issue])
  );
  dv.paragraph(`⚠️ 共 ${results.length} 个 KP 需要补充题库链接或题目`);
}
```

> **使用说明**：
> - 若输出"❌ 无题库关联"，可能原因：① 题库确实未拆分到该 KP；② 题目文件的 `knowledge_points` 未填写该 KP 名称；③ KP 正文中无 `04-题库/` 的 wikilink。
> - Agent 9 审计时，此项标记为 🟢 Info（非阻塞），用于提醒批量建题或补链接。

## 二、图片覆盖率缺口（Agent 9 · #13）

> **逻辑**：高优先级的 KP 必须配图；所有主版本已升级到 v1.3 的 KP，其 `images_note` 必须非空（已评估）。

### 2.1 高优待补（Dataview）

```dataview
TABLE
  subject AS 科目,
  images_priority AS 优先级,
  has_images AS 已配图,
  images_note AS 评估备注
FROM "03-知识点"
WHERE type = "知识点"
  AND images_priority = "high"
  AND has_images != true
SORT subject ASC
```

### 2.2 未评估债务（DataviewJS）

```dataviewjs
// 图片未评估债务：主版本 v1.3（兼容旧 template_version）且 images_note 为空
const kps = dv.pages('"03-知识点"').where(p =>

  dv.table(
    ["知识点", "subject", "images_priority", "状态"],
    kps.map(p => [
```

## 三、前置知识依赖链断裂（Agent 9 · #18）

> **逻辑**：KP 的 `prerequisite` 中的每个 wikilink 必须指向存在的文件，且目标 KP 的 status 应为 `已填充`（或至少是骨架以上）。
> 
> **限制**：Dataview 无法直接读取 prerequisite 数组中的 wikilink 并验证文件存在性（因为 prerequisite 存的是字符串数组，不是链接对象）。以下采用「间接统计法」：检查 KP 的入链/出链比例，辅助判断。

### 3.1 间接筛查：无前置出链的 KP（Dataview）

正常 KP 应该有前置知识出链。若 `file.outlinks` 极少，可能 prerequisite 填写了但未建立有效链接。

```dataview
TABLE
  subject AS 科目,
  length(prerequisite) AS 声明前置数,
  length(file.outlinks) AS 实际出链数,
  prerequisite AS 前置列表
FROM "03-知识点"
WHERE type = "知识点"
  AND length(prerequisite) > 0
  AND length(file.outlinks) < length(prerequisite)
SORT 实际出链数 ASC
```

> **解读**：`声明前置数 > 实际出链数` 说明 prerequisite 中写了一些不存在的文件名，或写了文件名但未用 `[[ ]]` 包裹（导致 Obsidian 不识别为链接）。

### 3.2 精确版（Node.js 脚本草案）

```javascript
// prerequisite-audit.js —— 建议由 Agent 9 在 Node 环境执行
// 前置：需要 node 环境和 vault 路径

const fs = require('fs');
const path = require('path');
const yaml = require('js-yaml'); // npm install js-yaml

const VAULT = "C:/Obsidion/妙妙屋"; // 根据实际路径调整
const KP_DIR = path.join(VAULT, "03-知识点");

function extractFrontmatter(content) {
  const match = content.match(/^---\n([\s\S]*?)\n---/);
  if (!match) return null;
  try {
    return yaml.load(match[1]);
  } catch (e) {
    return null;
  }
}

function kpExists(name) {
  // 在 KP_DIR 及其子目录中查找 <name>.md
  const dirs = fs.readdirSync(KP_DIR, { withFileTypes: true })
    .filter(d => d.isDirectory())
    .map(d => d.name);
  dirs.push("."); // 根目录
  for (const dir of dirs) {
    const fp = path.join(KP_DIR, dir, `${name}.md`);
    if (fs.existsSync(fp)) return fp;
  }
  return null;
}

const issues = [];

// 遍历所有 KP
function scan() {
  const dirs = fs.readdirSync(KP_DIR, { withFileTypes: true })
    .filter(d => d.isDirectory())
    .map(d => d.name);
  dirs.push(".");

  for (const dir of dirs) {
    const subDir = path.join(KP_DIR, dir);
    const files = fs.readdirSync(subDir).filter(f => f.endsWith('.md'));
    for (const file of files) {
      const content = fs.readFileSync(path.join(subDir, file), 'utf8');
      const fm = extractFrontmatter(content);
      if (!fm || !fm.prerequisite) continue;

      for (const pre of fm.prerequisite) {
        // 去除 wikilink 括号
        const cleanName = pre.replace(/\[\[(.*?)\]\]/, '$1').split('|')[0].trim();
        const exists = kpExists(cleanName);
        if (!exists) {
          issues.push({ kp: `${dir}/${file}`, prerequisite: pre, issue: "文件不存在" });
        } else {
          // 检查目标 status
          const targetContent = fs.readFileSync(exists, 'utf8');
          const targetFm = extractFrontmatter(targetContent);
          if (targetFm && targetFm.status !== "已填充") {
            issues.push({ kp: `${dir}/${file}`, prerequisite: pre, issue: `目标 status=${targetFm.status}` });
          }
        }
      }
    }
  }
}

scan();
console.log(`扫描完成，发现 ${issues.length} 处前置依赖问题：`);
issues.forEach(i => console.log(`- ${i.kp} → ${i.prerequisite} | ${i.issue}`));
```

> **部署建议**：
> - 将上述脚本保存为 `.claude/prerequisite-audit.js` 或 `scripts/prerequisite-audit.js`
> - Agent 9 健康检查时，若 Node 环境可用，执行此脚本获取精确结果
> - 若仅在 Obsidian 内运行，使用 §3.1 的 Dataview 间接筛查即可

## 四、教学就绪度统计（Agent 9 · #12 辅助）

> **逻辑**：统计 `teaching_ready` 字段分布，辅助评估知识库"使用成熟度"。

```dataview
TABLE
  length(rows) AS KP总数,
  length(filter(rows, (r) => r.teaching_ready = true)) AS "✅ 教学就绪",
  length(filter(rows, (r) => r.teaching_ready != true)) AS "❌ 未就绪",
  round(length(filter(rows, (r) => r.teaching_ready = true)) / length(rows) * 100, 1) + "%" AS 就绪率
FROM "03-知识点"
WHERE type = "知识点"
GROUP BY subject
SORT 就绪率 DESC
```

> **说明**：由于 `teaching_ready` 是 v1.3.1 新增字段，旧版 KP 可能无此字段（Dataview 中显示为空，归入"未就绪"）。建议在批量迁移时随主版本字段 `version` 一并统一补齐；兼容期内保留 `template_version` 不影响本统计。

## 五、组合仪表板（Agent 9 报告用）

以下查询可直接嵌入 `09-审计报告/<日期>-健康检查.md` 的对应章节：

```dataviewjs
// Agent 9 v1.2 组合摘要（一段代码出四个指标）
const kps = dv.pages('"03-知识点"').where(p => p.type == "知识点");
const problems = dv.pages('"04-题库"').where(p => p.type == "题目");

// 1. 图片已评估率
const v13Kps = kps.where(p => (p.version && p.version == "v1.3") || (!p.version && p.template_version == "v1.3"));
const assessedCount = v13Kps.where(p => p.images_note && p.images_note.toString().trim() !== "").length;
const assessRate = v13Kps.length > 0 ? Math.round(assessedCount / v13Kps.length * 100) : 0;

// 2. 高优待补
const highPriorityNoImage = kps.where(p => p.images_priority == "high" && p.has_images != true).length;

// 3. 题型覆盖率（声明了题型的KP中，有关联题目的比例）
const kpWithTypes = kps.where(p => p.problem_types && p.problem_types.length > 0);
// 近似：检查出链中是否包含 04-题库
let linkedKpCount = 0;
for (const kp of kpWithTypes) {
  const hasLink = kp.file.outlinks.some(l => l.path.includes("04-题库"));
  if (hasLink) linkedKpCount++;
}
const typeCoverage = kpWithTypes.length > 0 ? Math.round(linkedKpCount / kpWithTypes.length * 100) : 0;

// 4. 教学就绪率
const readyCount = kps.where(p => p.teaching_ready === true).length;
const readyRate = kps.length > 0 ? Math.round(readyCount / kps.length * 100) : 0;

dv.paragraph(`
| 指标 | 数值 | 说明 |
|:---|:---|:---|
| 图片已评估率 | ${assessRate}% (${assessedCount}/${v13Kps.length}) | 主版本为 v1.3 的 KP 中 images_note 非空的比例 |
| 高优待补 | ${highPriorityNoImage} 个 | images_priority=high 且未配图 |
| 题型覆盖率 | ${typeCoverage}% (${linkedKpCount}/${kpWithTypes.length}) | 声明题型的 KP 中有题库关联的比例 |
| 教学就绪率 | ${readyRate}% (${readyCount}/${kps.length}) | teaching_ready=true 的 KP 比例 |
`);
```
