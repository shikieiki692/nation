> **数据源**：`validate_kb.py` 每次运行自动记录到 `02-数据库/quality-metrics.jsonl`

```dataviewjs
const metricsFile = await dv.io.load("02-数据库/quality-metrics.jsonl");
const lines = metricsFile.trim().split("\n").filter(l => l.length > 0);
const records = lines.map(l => JSON.parse(l));

// 找最新一次 quick 模式记录（warnings < 1000 的视为 quick 模式）
const quickRecords = records.filter(r => r.warnings < 1000);
const latest = quickRecords[quickRecords.length - 1];

if (!latest) {
  dv.paragraph("⚠️ 尚未有 quick 模式扫描记录。运行 `python validate_kb.py --quick` 生成。");
} else {
  dv.header(3, `📊 最新扫描 · ${latest.date}`);
  
  const healthItems = [
    { label: "📁 受检文件", value: latest.files_checked, badge: "基准", good: true },
    { label: "🔴 错误 (Error)", value: latest.errors, badge: "需修复", good: latest.errors === 0 },
    { label: "🟡 警告 (Warning)", value: latest.warnings, badge: "需关注", good: latest.warnings < 100 },
    { label: "ℹ️ 信息 (Info)", value: latest.infos, badge: "建议项", good: latest.infos < 100 },
  ];
  
  dv.table(
    ["指标", "数值", "状态"],
    healthItems.map(item => [
      item.label,
      item.value.toString(),
      item.value === 0 ? "✅ 无" : item.good ? "✅ 良好" : `⚠️ ${item.badge}（${item.value}）`
    ])
  );
  
  dv.paragraph(`📄 详细报告：[[09-审计报告/auto-validation/${latest.date}-validation.md|${latest.date} 验证报告]]`);
}
```

## 📈 趋势（最近 5 次快速扫描）

```dataviewjs
const metricsFile = await dv.io.load("02-数据库/quality-metrics.jsonl");
const lines = metricsFile.trim().split("\n").filter(l => l.length > 0);
const records = lines.map(l => JSON.parse(l));
const quickRecords = records.filter(r => r.warnings < 1000 || r.warnings === 0);

if (quickRecords.length < 2) {
  dv.paragraph("⚠️ 至少需要 2 次扫描才能显示趋势。");
} else {
  const last5 = quickRecords.slice(-5);
  
  dv.table(
    ["日期", "🔴 Errors", "🟡 Warnings", "ℹ️ Info", "错误趋势"],
    last5.map(r => {
      const prevIdx = last5.indexOf(r) - 1;
      let dir = "—";
      if (prevIdx >= 0) {
        const prev = last5[prevIdx];
        dir = r.errors < prev.errors ? "📉" : r.errors > prev.errors ? "📈" : "➡️";
      }
      return [r.date, r.errors.toString(), r.warnings.toString(), (r.infos || 0).toString(), dir];
    })
  );
  
  // 整体方向
  const first = last5[0];
  const last = last5[last5.length - 1];
  const dirIcon = last.errors < first.errors ? "📉" : last.errors > first.errors ? "📈" : "➡️";
  dv.paragraph(`整体趋势：${dirIcon} **${first.errors} → ${last.errors}**（错误 ${last.errors < first.errors ? "减少" : last.errors > first.errors ? "增加" : "持平"}）`);
}
```

## 📋 生命周期分布（迁移后）

```dataviewjs
const kpFiles = dv.pages('"03-知识点"');
const handoutFiles = dv.pages('"04-课件/学生讲义"');
const outlineFiles = dv.pages('"04-课件/备课大纲"');
const topicFiles = dv.pages('"04-专题与题型/专题"');

function countStage(files) {
  const stages = {};
  for (const f of files) {
    const s = f.stage || "未设置";
    stages[s] = (stages[s] || 0) + 1;
  }
  return stages;
}

function renderTable(stages, title) {
  const entries = Object.entries(stages).sort((a, b) => b[1] - a[1]);
  const total = entries.reduce((s, [,n]) => s + n, 0);
  dv.header(4, `${title}（共 ${total}）`);
  dv.table(
    ["Stage", "数量", "占比"],
    entries.map(([s, n]) => [
      s === "未设置" ? `— ${s}` : s,
      n.toString(),
      (n / total * 100).toFixed(0) + "%"
    ])
  );
}

renderTable(countStage(kpFiles), "📚 知识点");
renderTable(countStage(outlineFiles), "📝 备课大纲");
renderTable(countStage(topicFiles), "📌 专题页");
renderTable(countStage(handoutFiles), "📖 学生讲义");
```

## 📊 分目录健康扫描

```dataviewjs
const dirs = [
  { name: "📚 知识点", path: '"03-知识点"' },
  { name: "📖 学生讲义", path: '"04-课件/学生讲义"' },
  { name: "📝 备课大纲", path: '"04-课件/备课大纲"' },
  { name: "📌 专题页", path: '"04-专题与题型/专题"' },
  { name: "💡 教学洞察", path: '"12-教学洞察"' },
  { name: "📄 资料提炼", path: '"07-资料提炼"' },
];

const rows = dirs.map(d => {
  const pages = dv.pages(d.path);
  const total = pages.length;
  const noTitle = pages.filter(p => !p.title).length;
  const noUpdated = pages.filter(p => !p.updated).length;
  return [d.name, total.toString(), noTitle.toString(), noUpdated.toString()];
});

dv.header(3, "🔍 覆盖度与完整性");
dv.table(
  ["目录", "文件数", "缺 title", "缺 updated"],
  rows
);
```

| 问题类型 | 当前规模 | 自动修复 | 入口 |
