> 排除：考纲条目互引、模块/总览页、题型/专题/错题占位、已有 KP 文件名子串匹配（假性缺失）。
> 当前口径：真正缺失 189 → **0**。其中 2026-05-17 Phase 5 结束时曾短暂收敛到 **4 个**（糖类 3 + 肽键 1）；截至 2026-06-03 已进一步闭环为 0。下方 DataviewJS 保留为持续监控工具，若再次出现结果，以扫描结果为准并回填本页策略段落。

```dataviewjs
// 反向扫描 02-考纲条目 中所有 wikilink，列出指向不存在文件的链接目标
const syllabus = dv.pages('"02-考纲条目"');

// 排除模式：考纲条目互引 / 模块 / 题型 / 专题 / 总览 / 资料来源
const excludeRegex = /^(中国化学奥林匹克(基本|决赛)要求-总览|(基础要求|决赛要求)-|(决赛)?\d{1,2}-|题型-|专题-|错题|外部资料-|提炼-|Arrow|ABOC|Atkins|物理化学（第)/;

// 专题/总结/比较/入门/应用等占位后缀（非基础KP，属于待建专题页）
// v1.3 新增："专题"和"总论"也是典型的专题/汇总占位命名
const topicSuffixRegex = /(专题|总论|总结|比较|入门|联系|应用|分析|初步|基础|模型|公式|策略|技巧)$/;

// 预先收集 03-知识点 下所有文件名（用于模糊匹配，筛除"假性缺失"）
// 使用 Obsidian 原生 API 绕过 Dataview 索引滞后问题
const kpFileNames = app.vault.getMarkdownFiles()
  .filter(f => f.path.replace(/\\/g, '/').startsWith('03-知识点/'))
  .map(f => f.basename);

const gaps = new Map(); // 真正缺失目标 → [来源条目链接]
const fuzzyGaps = new Map(); // 假性缺失（文件名子串匹配）→ [来源条目链接]
let skippedExists = 0;
let skippedExclude = 0;
let skippedTopic = 0;

for (const f of syllabus) {
  for (const link of f.file.outlinks) {
    const normalizedPath = link.path.replace(/\\/g, '/');
    const targetName = normalizedPath.split("/").pop().replace(/\.md$/, '');

    // 多层存在性检查（兼容 Windows 路径格式差异）
    function checkExists(p) {
      if (dv.page(p)) return true;
      if (dv.page(p.replace(/\.md$/, ''))) return true;
      const variants = [p];
      if (!p.endsWith('.md')) variants.push(p + '.md');
      if (p.endsWith('.md')) variants.push(p.slice(0, -3));
      variants.push(p.replace(/\//g, '\\'));
      variants.push(p.replace(/\\/g, '/'));
      for (const v of variants) {
        if (app.vault.getAbstractFileByPath(v)) return true;
        if (app.metadataCache.getFirstLinkpathDest(v, f.file.path)) return true;
      }
      return false;
    }
    if (checkExists(link.path)) { skippedExists++; continue; }

    if (excludeRegex.test(targetName)) { skippedExclude++; continue; }
    if (topicSuffixRegex.test(targetName)) { skippedTopic++; continue; }

    // 模糊匹配：缺失目标名是否是某个已有 KP 文件名的子串（或包含某个文件名）
    // 这能捕获「分散相」→「分散相与连续相」、「电极电势」→「标准电极电势」等假性缺失
    const fuzzyMatch = kpFileNames.find(name =>
      (targetName.length >= 3 && name.includes(targetName)) ||
      (name.length >= 3 && targetName.includes(name))
    );
    if (fuzzyMatch) {
      if (!fuzzyGaps.has(targetName)) fuzzyGaps.set(targetName, { sources: [], match: fuzzyMatch });
      fuzzyGaps.get(targetName).sources.push(f.file.link);
      continue;
    }

    if (!gaps.has(targetName)) gaps.set(targetName, []);
    gaps.get(targetName).push(f.file.link);
  }
}

// 按被引次数降序排列
const rows = [...gaps.entries()].sort(
  (a, b) => b[1].length - a[1].length || a[0].localeCompare(b[0])
);

if (rows.length === 0) {
  dv.paragraph("✅ 所有从 02-考纲条目 出发的 KP 引用均已存在。");
} else {
  dv.table(
    ["缺失 KP", "被引次数", "来源考纲条目"],
    rows.map(([name, sources]) => [
      name,
      sources.length,
      sources.slice(0, 5).join("、") + (sources.length > 5 ? ` …+${sources.length - 5}` : "")
    ])
  );
}
```
