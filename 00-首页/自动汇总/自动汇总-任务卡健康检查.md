# 自动汇总 · 任务卡健康检查

> 本页只负责聚合 `00-首页/活跃任务/` 的任务卡状态与 frontmatter 异常。
> 任务意义判断、优先级解释和关闭决策仍回到 [[活跃任务]] 与具体任务卡处理。

## 状态快照

```dataview
TABLE
  length(rows) as 总数,
  length(filter(rows, (r) => r.status = "active")) as Active,
  length(filter(rows, (r) => r.status = "blocked")) as Blocked,
  length(filter(rows, (r) => r.status = "completed")) as Completed,
  length(filter(rows, (r) => r.status = "active" AND r.priority = "P1")) as "P1 Active"
FROM "00-首页/活跃任务"
WHERE type = "活跃任务卡"
```

## 异常清单

```dataviewjs
const pages = dv.pages('"00-首页/活跃任务"').where(p => p.type === "活跃任务卡");
const issues = [];

for (const p of pages) {
  const missing = [];
  if (!p.status) missing.push("缺 status");
  if (!p.priority) missing.push("缺 priority");
  if (!p.area) missing.push("缺 area");
  if (!p.source_notes || (Array.isArray(p.source_notes) && p.source_notes.length === 0)) missing.push("缺 source_notes");
  if (!p.related_notes || (Array.isArray(p.related_notes) && p.related_notes.length === 0)) missing.push("缺 related_notes");
  if (!p.evidence || (Array.isArray(p.evidence) && p.evidence.length === 0)) missing.push("缺 evidence");
  if (p.status === "completed" && !p.completed) missing.push("completed 缺完成日期");
  if (missing.length > 0) issues.push([p.file.link, missing.join("；"), p.updated ?? "-"]);
}

if (issues.length === 0) {
  dv.paragraph("✅ 当前任务卡 frontmatter 关键字段完整，未发现 completed 漏日期问题。");
} else {
  dv.table(["任务卡", "异常", "updated"], issues);
}
```
