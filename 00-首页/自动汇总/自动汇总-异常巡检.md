## 任务卡 frontmatter 异常

```dataviewjs
const pages = dv.pages('"00-首页/活跃任务"').where(p => p.type === "活跃任务卡");
const issues = [];

for (const p of pages) {
  const missing = [];
  if (!p.status) missing.push("缺 status");
  if (!p.priority) missing.push("缺 priority");
  if (!p.area) missing.push("缺 area");
  if (!p.source_notes || (Array.isArray(p.source_notes) && p.source_notes.length === 0)) missing.push("缺 source_notes");
  if (!p.related_notes || (Array.isArray(p.related_notes) && p.related_notes.length === 0)) missing.push("缺 related_notes");
  if (!p.evidence || (Array.isArray(p.evidence) && p.evidence.length === 0)) missing.push("缺 evidence");
  if (p.status === "completed" && !p.completed) missing.push("completed 缺完成日期");
  if (missing.length > 0) issues.push([p.file.link, missing.join("；"), p.updated ?? "-"]);
}

if (issues.length === 0) {
  dv.paragraph("✅ 当前任务卡未发现关键 frontmatter 异常。");
} else {
  dv.table(["任务卡", "异常", "updated"], issues);
}
```

## 系统页残留静态数字复核

> 本区只做“需人工复核”的轻提示，不直接判错。
> 只扫描少数系统入口页，避免把工作日志、历史页、设计备忘里的历史数字误报为异常。

```dataviewjs
const targets = [
  "00-首页/首页.md",
  "00-首页/状态摘要.md",
  "00-首页/活跃任务.md",
  "00-首页/Agent战略方向.md",
  "00-首页/Agent入口文件关系说明.md"
];

const ratioPattern = /\b\d+\/\d+\b/g;
const percentPattern = /\b100%\b/g;
const rows = [];

for (const path of targets) {
  const page = dv.page(path);
  const content = page?.file?.content;
  if (!content) continue;

  const lines = content.split(/\r?\n/);
  lines.forEach((line, index) => {
    if (line.includes("```")) return;
    if (line.includes("![[自动汇总-")) return;

    const ratioMatches = [...line.matchAll(ratioPattern)].map(m => m[0]);
    const percentMatches = line.includes("已填充") ? [...line.matchAll(percentPattern)].map(m => m[0]) : [];
    const matches = [...ratioMatches, ...percentMatches];

    if (matches.length === 0) return;

    rows.push([
      page.file.link,
      index + 1,
      matches.join("，"),
      line.trim().slice(0, 120)
    ]);
  });
}

if (rows.length === 0) {
  dv.paragraph("✅ 当前纳入扫描的系统入口页未发现明显需要人工复核的静态进度数字。");
} else {
  dv.table(["页面", "行号", "命中", "上下文"], rows);
}
```

## 系统页残留旧口径提示复核

> 本区优先抓“旧字段/旧版本口径仍在正文里被当作当前规则使用”的情况。
> 明确写成“兼容 / 退役 / 不再作为真相源”的说明，不视为异常。

```dataviewjs
const targets = [
  "00-首页/首页.md",
  "00-首页/状态摘要.md",
  "00-首页/活跃任务.md",
  "00-首页/Agent战略方向.md",
  "00-首页/Agent入口文件关系说明.md"
];

const suspiciousPatterns = [
  /template_version/g,
  /agent_prompts_version/g,
  /知识点模板 v1/g,
  /资料提炼模板 v1/g,
  /备课大纲 v1(?!\.)/g,
  /专题页 v1/g
];

const safeHints = ["兼容", "退役", "不再作为真相源", "历史", "旧字段", "逐步退役"];
const rows = [];

for (const path of targets) {
  const page = dv.page(path);
  const content = page?.file?.content;
  if (!content) continue;

  const lines = content.split(/\r?\n/);
  lines.forEach((line, index) => {
    if (safeHints.some(hint => line.includes(hint))) return;

    const hits = [];
    for (const pattern of suspiciousPatterns) {
      const matches = [...line.matchAll(pattern)].map(m => m[0]);
      if (matches.length > 0) hits.push(...matches);
    }

    if (hits.length === 0) return;

    rows.push([
      page.file.link,
      index + 1,
      [...new Set(hits)].join("，"),
      line.trim().slice(0, 120)
    ]);
  });
}

if (rows.length === 0) {
  dv.paragraph("✅ 当前纳入扫描的系统入口页未发现明显残留的旧口径提示。");
} else {
  dv.table(["页面", "行号", "命中", "上下文"], rows);
}
```
