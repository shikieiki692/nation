| 入口 | 什么时候先看它 | 它负责回答什么 | 冲突时怎么处理 |

// === 知识点统计 ===
const kpAll = dv.pages('"03-知识点"').where(p => p.type == "知识点");

const finalFilled = finalKP.where(p => p.status == "已填充").length;
const finalTotal = finalKP.length;

const basePct = baseTotal > 0 ? (baseFilled / baseTotal * 100).toFixed(0) : 0;

const modules = [
  { name: "化学原理", path: "03-知识点/化学原理" },
  { name: "无机", path: "03-知识点/无机和结构化学" },

  const mTotal = mKPs.length;

  moduleParts.push(`${m.name}${icon}`);

// === 最近工作日志 ===
const logFolder = "00-首页/工作日志";
const logFiles = app.vault.getMarkdownFiles()
  .filter(f => f.path.startsWith(logFolder + "/"))
  .sort((a, b) => b.stat.mtime - a.stat.mtime);

for (const file of logFiles) {
  const content = await app.vault.read(file);
  const lines = content.split("\n");
  for (const line of lines) {

**一句话理解各层职责**：
- 考纲层负责“边界与覆盖”，不承载正文主内容。
- 提炼层负责“把外部资料转成可消费资产”，尤其补上“怎么讲”的教学逻辑。
- KP / 专题 / 题库层负责“知识组织、解题组织、检索组织”。
- 备课层负责“教师怎么讲”，讲义层负责“学生实际看到什么”。
- 任务 / 日志 / 摘要 / 战略层负责“下一步做什么、做过什么、为什么这么做”。
