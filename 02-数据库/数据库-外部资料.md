## 一、已入库提炼页总览（低频参考）

```dataview
TABLE WITHOUT ID
  link(file.name, title) as 提炼页,
  source_type as 来源类型,
  year as 年份,
  subject as 学科范围,
  review_status as 审核状态,
  problems_planned as 计划题数,
  problems_extracted as 已提取,
  round(problems_extracted / problems_planned * 100, 1) + "%" as 提取进度,
  length(knowledge_points) as 涉及知识点数
FROM "07-资料提炼"
SORT review_status ASC, year DESC
```

## 三、考纲覆盖分析

> 以下统计各提炼页覆盖的考纲条目，用于检查是否有考纲盲区。

```dataview
TABLE WITHOUT ID
  syllabus_codes as 考纲条目,
  count(file.name) as 被引用次数,
  join(map(rows, (r) => link(r.file.name, r.title)), "、") as 来源提炼页
FROM "07-资料提炼"
FLATTEN syllabus_codes
GROUP BY syllabus_codes
SORT count(file.name) DESC
LIMIT 30
```

## 五、知识点入库追踪

> 检查提炼页中提到的知识点是否已在 `03-知识点/` 下创建对应文件。

```dataviewjs
// 获取所有提炼页中提到的 knowledge_points
const提炼Files = await app.vault.getMarkdownFiles()
  .filter(f => f.path.startsWith("07-资料提炼/"));

const提炼KP = new Set();
for (const f of 提炼Files) {
  const content = await app.vault.read(f);
  const fm = app.metadataCache.getFileCache(f)?.frontmatter;
  if (fm && fm.knowledge_points) {
    for (const kp of fm.knowledge_points) {
      提炼KP.add(kp);
    }
  }
}

// 获取所有已创建的知识点文件名
const kpFiles = await app.vault.getMarkdownFiles()
  .filter(f => f.path.startsWith("03-知识点/"));
const kpNames = new Set(kpFiles.map(f => f.basename));

// 找出尚未创建的知识点
const missing = [];
for (const kp of 提炼KP) {
  if (!kpNames.has(kp)) {
    missing.push(kp);
  }
}

if (missing.length === 0) {
  dv.paragraph("✅ 所有提炼页涉及的知识点均已创建文件。");
} else {
  dv.paragraph("⚠️ 以下知识点在提炼页中被提及，但尚未在 `03-知识点/` 下创建文件：");
  dv.list(missing.map(m => `[[${m}]]（待创建）`));
}
```

## 六、真题入库追踪

> 检查提炼页中计划的题目是否已拆分为 `04-题库/真题/` 下的独立文件。

```dataviewjs
const提炼Files = await app.vault.getMarkdownFiles()
  .filter(f => f.path.startsWith("07-资料提炼/"));

let totalPlanned = 0;
let totalExtracted = 0;

for (const f of 提炼Files) {
  const fm = app.metadataCache.getFileCache(f)?.frontmatter;
  if (fm) {
    totalPlanned += fm.problems_planned || 0;
    totalExtracted += fm.problems_extracted || 0;
  }
}

const pct = totalPlanned > 0 ? Math.round(totalExtracted / totalPlanned * 100) : 0;

dv.paragraph(`**真题提取总进度**：${totalExtracted} / ${totalPlanned} 题（${pct}%）`);
```

## 七、资料流水线状态图

```
┌─────────────────┐     ┌─────────────┐     ┌─────────────────┐     ┌─────────────┐
│ 06-外部资料导入  │ ──→ │ 07-资料提炼  │ ──→ │ 03-知识点/       │ ──→ │ 04-题库/真题 │
│ （原始文件）      │     │ （提炼页）    │     │ （知识点文件）    │     │ （题目文件）  │
└─────────────────┘     └─────────────┘     └─────────────────┘     └─────────────┘
       ↑                                                              ↑
       └────────────────── 新资料来源（教材/真题/论文）──────────────────┘
```

| 环节 | 当前数量 | 状态 |
|:---:|:---:|:---|
